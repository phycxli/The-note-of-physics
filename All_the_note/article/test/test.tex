\begin{equation*}
  (ye^{-\frac{x^2}{2}})'=e^{-\frac{x^2}{2}}(y'-xy)
\end{equation*}
\begin{equation*}
  y'-xy=\frac{1}{2\sqrt{x}}e^{\frac{x^2}{2}}
\end{equation*}
两边乘以$e^{-\frac{x^2}{2}}$
\begin{equation*}
  e^{-\frac{x^2}{2}}(y'-xy)=(ye^{-\frac{x^2}{2}})'=\frac{1}{2\sqrt{x}}
\end{equation*}
